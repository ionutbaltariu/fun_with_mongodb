// Creating tables
Table users{
  _id ObjectId [pk]
  first_name string
  last_name string
  user_name string
  email string
  registration_date string
  channels list[channel_Id]
}

Table channels {
  _id ObjectId [pk]
  name string
  type string
  creation_date datetime
}

Table comments {
  _id ObjectId [pk]
  value string
  channel channel_id
  user user_id
  sent_at datetime
}
